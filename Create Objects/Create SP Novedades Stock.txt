ALTER PROCEDURE SP_NOVEDADES_STOCK (IN  ItemCode NVARCHAR(100), IN  Leido NVARCHAR(1)
)
AS
BEGIN
    -- 2. Lógica de modificación (DML)
    UPDATE "ST_NOVEDADES_STOCK"
    SET "Leido" = :Leido
    WHERE "ItemCode" = :ItemCode
    ;

END;
